```json
{
  "topic": "realestate/contact#add",
  "source": "http://{user}.example.com/profile/card#me",

  "data": {
    "type": "AddAction",
    "actionStatus": "CompletedActionStatus",
    "agent": "http://{user}.example.com/profile/card#me",
    "instrument": "http://{app}.example.com/profile/card#me",

      "type": "Contact",
      "name": "Adam Addme"
    },
    "targetCollection": {
      "type": "Collection",
      "id": "https://{user}.example.com/data/contact/"
    }
  }
}
```
